lqk_print.jar


#Usage
	make sure these files exist
	* libs/lqk_printer.jar
	* assets/printer.json
	* libs/armeabi/libserial_port.so
	* libs/armeabi-v7a/libserial_port.so
	* libs/armeabi-v7a/liblqk_encrypt.so



SampleApp
	click "search" will search for all avaliable printers
	click a result printer will try to connect
	click a connected printer will print test content
	click "print" will print test content via all connected printers
